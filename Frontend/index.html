      ///
      /// Deleting the node without using the Head pointer in a single linked list
      /// for this problem we need to know the actually address of the required node
      /// hence we nedd to save that address by using a globle varible and that
      /// is in creatnode function..here i am saving the 3rd node address from last
      /////////////////////////////////////////////////////////////////////////
      #include
      #include

      struct node{
      int data;
      struct node *next;
      };
      struct node *ptr1;
      int count;
      ////creating nodes at first/////////////////////////////////////////////
      void creatnode(struct node **head,int data){
      struct node *newnode=(struct node*)malloc(sizeof(struct node));
      count++;
      if(count==3)
      ptr1=newnode;
      newnode->data=data;
      newnode->next=*head;
      *head=newnode;
      }
      //////Display the list of nodes/////////////////////////////////////////
      void display(struct node *head){
      if(head==NULL){
      printf(“List is Empty\n”);
      return;
      }
      //head is traversing till the end of the list
      while(head){
      printf(“%d\t”,head->data);
      //printf(“%p\t”,head);
      head=head->next;
      }
      }
      ///////////////////Deleting the node without using head pointer///////////
      //we need to copy the next node information into current node nad we are actually
      //deleting the next node(since we have the all the infoprmation of next node in cuurent node)
      void deletenode(struct node *ptr){
      //saving the next nod address
      struct node *temp=ptr->next;
      ptr->data=ptr->next->data;
      //printf(“%d\t”,ptr->data);
      ptr->next=ptr->next->next;
      //printf(“%p\t”,ptr->next);
      free(temp);
      }
      int main(){
      struct node *head=NULL;
      creatnode(&head,70);
      creatnode(&head,60);
      creatnode(&head,50);
      creatnode(&head,40);
      creatnode(&head,30);
      creatnode(&head,20);
      creatnode(&head,10);
      printf(“Display the nodes before deletation\n”);
      display(head);
      /////calling the deletenode function with saved argument ptr1/////
      deletenode(ptr1);
      printf(“\nDisplay the nodes after deletation\n”);
      display(head);

      return 0;
      }
